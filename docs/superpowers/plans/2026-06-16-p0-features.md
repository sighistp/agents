```python
# blueprint/utils/cost_tracker.py
"""Cost tracker — records LLM token usage per project."""
import contextvars
from typing import Any

# DeepSeek v4 flash pricing (USD per 1M tokens)
PRICING = {
    "input": 0.27,
    "output": 1.10,
}

# Context variable for concurrent project isolation
_current_project_id: contextvars.ContextVar[str] = contextvars.ContextVar('project_id', default='')


def set_project_context(project_id: str):
    """Set current project ID for cost tracking (via contextvars)."""
    _current_project_id.set(project_id)


class CostTracker:
    """In-memory cost tracker. Thread-safe via per-instance state."""

    def __init__(self):
        self._costs: dict[str, dict[str, Any]] = {}

    def record_cost(self, response, project_id: str = "") -> None:
        """Extract token usage from LLM response and record it."""
        # Prefer passed project_id, fallback to contextvars
        pid = project_id or _current_project_id.get()
        if not pid:
            return
        try:
            usage = getattr(response, "usage_metadata", None)
            if not usage:
                return
            input_tokens = usage.get("input_tokens", 0) or 0
            output_tokens = usage.get("output_tokens", 0) or 0
            if input_tokens == 0 and output_tokens == 0:
                return

            if pid not in self._costs:
                self._costs[pid] = {
                    "project_id": pid,
                    "total_calls": 0,
                    "total_input_tokens": 0,
                    "total_output_tokens": 0,
                    "estimated_cost_usd": 0.0,
                }
            data = self._costs[pid]
            data["total_calls"] += 1
            data["total_input_tokens"] += input_tokens
            data["total_output_tokens"] += output_tokens
            data["estimated_cost_usd"] = round(
                data["total_input_tokens"] / 1_000_000 * PRICING["input"]
                + data["total_output_tokens"] / 1_000_000 * PRICING["output"],
                6,
            )
        except Exception:
            pass  # Never crash

    def get_cost(self, project_id: str) -> dict:
        """Get cost data for a project. Returns empty dict if not found."""
        return self._costs.get(project_id, {})


# Module-level singleton
_tracker = CostTracker()


def record_cost(response, project_id: str = "") -> None:
    """Module-level convenience function."""
    _tracker.record_cost(response, project_id)


def get_cost(project_id: str) -> dict:
    """Module-level convenience function."""
    return _tracker.get_cost(project_id)
```

```python
def _extract_test_passed(summary: str, had_execution_errors: bool) -> bool:
    """Extract test result from summary using structured parsing.

    Priority:
    1. Execution errors → False
    2. Regex extraction of passed/failed/error counts
    3. Keyword fallback
    """
    if had_execution_errors:
        return False

    import re
    summary_lower = summary.lower()

    # Try regex extraction first (most reliable)
    passed_match = re.search(r'(\d+)\s*passed', summary_lower)
    failed_match = re.search(r'(\d+)\s*failed', summary_lower)
    error_match = re.search(r'(\d+)\s*error', summary_lower)

    if failed_match and int(failed_match.group(1)) > 0:
        return False
    if error_match and int(error_match.group(1)) > 0:
        return False
    if passed_match and int(passed_match.group(1)) > 0:
        return True

    # Keyword fallback
    pass_keywords = ["全部通过", "all passed", "all tests passed",
                     "0 fail", "0 error", "0失败", "0 failed"]
    for kw in pass_keywords:
        if kw in summary_lower:
            return True

    return True  # Default to passed
```
